# 举例使用

## 1. 基础使用

```go
package main

import (
    "log/slog"
    "backed/internal/logger"
)

func main() {
    // 最简初始化：JSON 格式输出到 stdout，Info 级别
    log := logger.New()
    log.Info("服务启动", "port", 8080)
    // 输出: {"time":"...","level":"INFO","msg":"服务启动","port":8080}

    // 开发环境：Text 格式 + Debug 级别，便于阅读
    devLog := logger.New(
        logger.WithFormat(logger.FormatText),
        logger.WithLevel(slog.LevelDebug),
    )
    devLog.Debug("加载配置", "path", "/etc/app.yml")
    // 输出: time=... level=DEBUG msg="加载配置" path=/etc/app.yml
}
```

## 2. 各种日志级别

```go
log := logger.New(logger.WithLevel(slog.LevelDebug))

log.Debug("调试信息", "step", 1)
log.Info("常规信息", "users", 42)
log.Warn("警告", "memory_mb", 850)
log.Error(errors.New("连接超时"), "数据库查询失败", "table", "users", "retry", 3)
```

## 3. 上下文日志 — 带 context 的调用

```go
// ctx 中携带 trace_id，由 ContextExtractor 自动提取并注入日志
ctx := context.WithValue(context.Background(), "trace_id", "abc-123")

log := logger.New(logger.WithContextExtractor(myExtractor))
log.InfoContext(ctx, "处理请求", "method", "GET")
log.ErrorContext(ctx, err, "处理失败", "endpoint", "/api/users")
```

## 4. 错误自动堆栈

```go
log := logger.New(
    logger.WithStackOnError(true), // 默认开启
)

if err := doWork(); err != nil {
    log.Error(err, "任务执行失败")
    // 输出自动包含:
    //   "error": "xxx"
    //   "stacktrace": "goroutine 1 [running]:\nmain.doWork()\n\t..."
}
```

## 5. 敏感数据脱敏

```go
log := logger.New()

// 直接使用 SensitiveString，输出 [REDACTED]
log.Info("用户认证", "token", logger.SensitiveString("sk-secret-token-xxx"))
// {"level":"INFO","msg":"用户认证","token":"[REDACTED]"}

// 自定义脱敏：包装结构体
type SafeUser struct{ logger.MaskedJSON }

func (u SafeUser) LogValue() slog.Value {
    user := u.Data.(User)
    user.Password = "***"
    return slog.AnyValue(user)
}

log.Info("用户登录", "user", SafeUser{Data: user})
```

## 6. 多输出 — 同时写控制台和文件

```go
f, _ := os.OpenFile("/var/log/cybertown.log", os.O_CREATE|os.O_WRONLY|os.O_APPEND, 0666)
defer f.Close()

log := logger.New(
    logger.WithFormat(logger.FormatJSON),
    logger.WithMultiWriter(
        slog.NewJSONHandler(os.Stdout, &slog.HandlerOptions{AddSource: true}),
        slog.NewJSONHandler(f, nil),
    ),
)

log.Info("多输出测试")
// 控制台和文件同时写入
```

## 7. 高流量采样

```go
// 生产环境：只保留 10% 的 Info 级别日志
log := logger.New(
    logger.WithSampling(0.1),
)

// 每秒 10 万条日志 → 实际写入约 1 万条
for i := 0; i < 100_000; i++ {
    log.Info("高频事件", "seq", i)
}
```

## 8. 键名转换 + 字段过滤

```go
log := logger.New(
    logger.WithReplaceAttr(func(groups []string, a slog.Attr) slog.Attr {
        // 丢弃密码字段
        if a.Key == "password" {
            return slog.Attr{}
        }
        // 丢弃空值
        if a.Value.Kind() == slog.KindString && a.Value.String() == "" {
            return slog.Attr{}
        }
        return a
    }),
)

log.Info("请求", "username", "alice", "password", "secret", "empty", "")
// password 和 empty 不会出现在输出中
```

## 9. CyberTown 业务语义 — AppLogger

```go
import (
    "context"
    "backed/internal/logger"
)

func main() {
    // 开发环境
    log := logger.NewApp("debug", true)

    ctx := context.Background()

    // NPC 日志 — 自动携带 npc_id
    log.NPC(ctx, "npc_42", "NPC 开始巡逻", "zone", "market")

    // 玩家日志 — 自动携带 player_id
    log.Player(ctx, "player_alice", "玩家进入区域", "zone", "market")

    // 区域日志
    log.Zone(ctx, "market", "区域活动更新", "npc_count", 15, "player_count", 8)

    // 行为日志
    log.Action(ctx, "trade_001", "交易完成", "seller", "npc_42", "buyer", "player_alice")
}
```

## 10. CyberTown 完整服务示例

```go
package main

import (
    "context"
    "errors"
    "log/slog"
    "os"

    "backed/internal/logger"
)

func main() {
    // ---------- 初始化日志 ----------
    // 开发环境：Text 格式 + Debug 级别
    log := logger.NewApp("debug", true)

    // ---------- 生产环境写法（注释示例）----------
    // log := logger.NewApp("info", false)
    // 如需同时写文件：
    // f, _ := os.OpenFile("/var/log/cybertown.log", os.O_CREATE|os.O_WRONLY|os.O_APPEND, 0666)
    // prodLog := logger.New(
    //     logger.WithLevel(slog.LevelInfo),
    //     logger.WithFormat(logger.FormatJSON),
    //     logger.WithSampling(0.1),
    //     logger.WithMultiWriter(
    //         slog.NewJSONHandler(os.Stdout, nil),
    //         slog.NewJSONHandler(f, nil),
    //     ),
    //     logger.WithContextExtractor(otelExtractor),
    // )

    // ---------- 服务启动 ----------
    log.Info("CyberTown 服务启动", "version", "1.0.0")

    // ---------- 模拟 NPC 行为 ----------
    ctx := context.Background()

    // NPC 巡逻
    for _, npc := range []string{"npc_merchant", "npc_guard", "npc_bard"} {
        log.NPC(ctx, npc, "NPC 进入巡逻状态")
    }

    // NPC 交互
    log.NPC(ctx, "npc_merchant", "商人摆出商品", "items", 12)
    log.NPCDebug(ctx, "npc_merchant", "商品价格计算完成", "total", 156)

    // ---------- 模拟玩家行为 ----------
    log.Player(ctx, "player_alice", "玩家连接成功", "ip", "192.168.1.100")
    log.Player(ctx, "player_bob", "玩家连接成功", "ip", "192.168.1.200")

    // 玩家进入区域
    log.Zone(ctx, "market_square", "区域加载完成", "buildings", 24)
    log.Player(ctx, "player_alice", "玩家进入区域", "zone", "market_square")

    log.Action(ctx, "trade_001", "交易发起", "item", "sword", "price", 100)
    log.Action(ctx, "trade_002", "交易完成",
        "buyer", "player_alice",
        "seller", "npc_merchant",
        "item", "sword",
    )

    // ---------- 错误处理 ----------
    if err := someRiskyOperation(); err != nil {
        // ErrorContext 自动携带堆栈和 npc_id
        log.NPCError(ctx, "npc_guard",
            errors.New("巡逻路径不可达"),
            "NPC 巡逻中断",
            "waypoint", 7,
        )
    }

    // ---------- 敏感信息保护 ----------
    log.Info("新玩家注册",
        "username", "alice",
        "token", logger.SensitiveString("sk-jwt-token-xxx"),
    )
}

func someRiskyOperation() error { return errors.New("模拟错误") }
```
