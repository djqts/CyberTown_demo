## 2. 封装核心

### **logger.go** —— 核心类型与构造函数

```go
package logger

import (
	"context"
	"io"
	"log/slog"
	"os"

	slogmulti "github.com/samber/slog-multi"
	slogsampling "github.com/samber/slog-sampling"
)

// Logger 是对 slog.Logger 的增强封装
type Logger struct {
	*slog.Logger
	config Config
}

// New 根据配置选项创建一个新的 Logger
func New(opts ...Option) *Logger {
	cfg := defaultConfig()
	for _, opt := range opts {
		opt(&cfg)
	}

	var handler slog.Handler
	baseHandler := cfg.handler

	// 如果未显式设置 handler，根据输出和格式创建
	if baseHandler == nil {
		w := cfg.writer
		if w == nil {
			w = os.Stdout
		}
		ho := &slog.HandlerOptions{
			Level:       cfg.level,
			AddSource:   cfg.addSource,
			ReplaceAttr: cfg.replaceAttr,
		}
		switch cfg.format {
		case FormatJSON:
			baseHandler = slog.NewJSONHandler(w, ho)
		default:
			baseHandler = slog.NewTextHandler(w, ho)
		}
	}

	// 堆栈中间件：错误时自动附加堆栈
	if cfg.stackOnError {
		baseHandler = newStacktraceMiddleware(baseHandler)
	}

	// 上下文中间件：从 context 中提取字段
	if cfg.contextExtractor != nil {
		baseHandler = newContextHandler(baseHandler, cfg.contextExtractor)
	}

	// 采样中间件
	if cfg.sampling.enabled {
		baseHandler = slogmulti.
			Pipe(cfg.sampling.option.NewMiddleware()).
			Handler(baseHandler)
	}

	// 多 handler 支持（例如同时写文件与 stdout）
	if len(cfg.multiWriters) > 0 {
		handlers := []slog.Handler{baseHandler}
		for _, w := range cfg.multiWriters {
			handlers = append(handlers, w)
		}
		handler = slogmulti.Fanout(handlers...)
	} else {
		handler = baseHandler
	}

	return &Logger{
		Logger: slog.New(handler),
		config: cfg,
	}
}

// Enhanced 方法：带上下文的便捷方法
func (l *Logger) DebugContext(ctx context.Context, msg string, args ...any) {
	l.Logger.DebugContext(ctx, msg, args...)
}

func (l *Logger) InfoContext(ctx context.Context, msg string, args ...any) {
	l.Logger.InfoContext(ctx, msg, args...)
}

func (l *Logger) WarnContext(ctx context.Context, msg string, args ...any) {
	l.Logger.WarnContext(ctx, msg, args...)
}

// ErrorContext 自动附加错误对象，错误字段会包含堆栈
func (l *Logger) ErrorContext(ctx context.Context, err error, msg string, args ...any) {
	attrs := make([]any, 0, len(args)+1)
	attrs = append(attrs, slog.Any("error", err))
	attrs = append(attrs, args...)
	l.Logger.ErrorContext(ctx, msg, attrs...)
}

// 无上下文的简写
func (l *Logger) Debug(msg string, args ...any) {
	l.DebugContext(context.Background(), msg, args...)
}

func (l *Logger) Info(msg string, args ...any) {
	l.InfoContext(context.Background(), msg, args...)
}

func (l *Logger) Warn(msg string, args ...any) {
	l.WarnContext(context.Background(), msg, args...)
}

func (l *Logger) Error(err error, msg string, args ...any) {
	l.ErrorContext(context.Background(), err, msg, args...)
}
```

### **config.go** —— 灵活的配置选项

```go
package logger

import (
	"io"
	"log/slog"

	slogsampling "github.com/samber/slog-sampling"
)

// Format 常量
const (
	FormatText = "text"
	FormatJSON = "json"
)

// Config 容纳所有配置
type Config struct {
	level       slog.Leveler
	format      string
	writer      io.Writer
	handler     slog.Handler
	addSource   bool
	stackOnError bool

	sampling struct {
		enabled bool
		option  slogsampling.UniformSamplingOption
	}

	contextExtractor func(ctx context.Context) []slog.Attr
	replaceAttr     func(groups []string, a slog.Attr) slog.Attr

	multiWriters []slog.Handler // 额外输出
}

func defaultConfig() Config {
	return Config{
		level:         slog.LevelInfo,
		format:        FormatJSON,
		addSource:     true,
		stackOnError: true,
	}
}

type Option func(*Config)

// WithLevel 设置日志级别
func WithLevel(level slog.Leveler) Option {
	return func(c *Config) { c.level = level }
}

// WithFormat 设置输出格式 (json/text)
func WithFormat(f string) Option {
	return func(c *Config) { c.format = f }
}

// WithWriter 设置输出目标 (默认 os.Stdout)
func WithWriter(w io.Writer) Option {
	return func(c *Config) { c.writer = w }
}

// WithHandler 直接注入 slog.Handler (优先级高于 format/writer)
func WithHandler(h slog.Handler) Option {
	return func(c *Config) { c.handler = h }
}

// WithSource 启用源码位置记录
func WithSource(b bool) Option {
	return func(c *Config) { c.addSource = b }
}

// WithStackOnError 是否在 Error 日志中自动附加堆栈
func WithStackOnError(b bool) Option {
	return func(c *Config) { c.stackOnError = b }
}

// WithSampling 启用均匀采样，rate 范围 0.0~1.0
func WithSampling(rate float64) Option {
	return func(c *Config) {
		c.sampling.enabled = true
		c.sampling.option = slogsampling.UniformSamplingOption{
			Rate: rate,
		}
	}
}

// WithContextExtractor 设置从 context 中提取字段的函数
func WithContextExtractor(fn func(ctx context.Context) []slog.Attr) Option {
	return func(c *Config) {
		c.contextExtractor = fn
	}
}

// WithReplaceAttr 自定义字段处理（如脱敏、键名转换）
func WithReplaceAttr(fn func(groups []string, a slog.Attr) slog.Attr) Option {
	return func(c *Config) {
		c.replaceAttr = fn
	}
}

// WithMultiWriter 同时写入多个 handler（例如文件 + stdout）
func WithMultiWriter(handlers ...slog.Handler) Option {
	return func(c *Config) {
		c.multiWriters = append(c.multiWriters, handlers...)
	}
}

// 便捷方法：直接写入文件并保持 JSON 格式
func WithFileOutput(path string) Option {
	return func(c *Config) {
		f, err := os.OpenFile(path, os.O_CREATE|os.O_WRONLY|os.O_APPEND, 0666)
		if err != nil {
			panic("failed to open log file: " + err.Error())
		}
		c.writer = f
	}
}
```

### **stacktrace.go** —— 自动堆栈注入中间件

```go
package logger

import (
	"context"
	"log/slog"
	"runtime/debug"
)

// stacktraceMiddleware 在日志级别为 Error 且包含 "error" 字段时自动添加堆栈
type stacktraceMiddleware struct {
	next slog.Handler
}

func newStacktraceMiddleware(next slog.Handler) slog.Handler {
	return &stacktraceMiddleware{next: next}
}

func (h *stacktraceMiddleware) Enabled(ctx context.Context, level slog.Level) bool {
	return h.next.Enabled(ctx, level)
}

func (h *stacktraceMiddleware) Handle(ctx context.Context, r slog.Record) error {
	if r.Level == slog.LevelError {
		// 检查是否已经存在 stacktrace 字段，避免重复添加
		hasStack := false
		r.Attrs(func(a slog.Attr) bool {
			if a.Key == "stacktrace" {
				hasStack = true
				return false // 停止迭代
			}
			return true
		})
		if !hasStack {
			r.AddAttrs(slog.String("stacktrace", string(debug.Stack())))
		}
	}
	return h.next.Handle(ctx, r)
}

func (h *stacktraceMiddleware) WithAttrs(attrs []slog.Attr) slog.Handler {
	return &stacktraceMiddleware{next: h.next.WithAttrs(attrs)}
}

func (h *stacktraceMiddleware) WithGroup(name string) slog.Handler {
	return &stacktraceMiddleware{next: h.next.WithGroup(name)}
}
```

### **context.go** —— 上下文富化处理器（可选）

```go
package logger

import (
	"context"
	"log/slog"
)

type contextHandler struct {
	next     slog.Handler
	extract  func(ctx context.Context) []slog.Attr
}

func newContextHandler(next slog.Handler, extract func(ctx context.Context) []slog.Attr) slog.Handler {
	return &contextHandler{next: next, extract: extract}
}

func (h *contextHandler) Enabled(ctx context.Context, level slog.Level) bool {
	return h.next.Enabled(ctx, level)
}

func (h *contextHandler) Handle(ctx context.Context, r slog.Record) error {
	if h.extract != nil {
		attrs := h.extract(ctx)
		r.AddAttrs(attrs...)
	}
	return h.next.Handle(ctx, r)
}

func (h *contextHandler) WithAttrs(attrs []slog.Attr) slog.Handler {
	return &contextHandler{next: h.next.WithAttrs(attrs), extract: h.extract}
}

func (h *contextHandler) WithGroup(name string) slog.Handler {
	return &contextHandler{next: h.next.WithGroup(name), extract: h.extract}
}
```

**使用示例**（配合 OpenTelemetry）：

```go
import "go.opentelemetry.io/otel/trace"

func otelContextExtractor(ctx context.Context) []slog.Attr {
	span := trace.SpanFromContext(ctx)
	if !span.IsRecording() {
		return nil
	}
	return []slog.Attr{
		slog.String("trace_id", span.SpanContext().TraceID().String()),
		slog.String("span_id", span.SpanContext().SpanID().String()),
	}
}
```

### **sensitive.go** —— 敏感信息脱敏（LogValuer）

```go
package logger

import "log/slog"

// SensitiveString 隐藏真实值，日志中显示为 [REDACTED]
type SensitiveString string

func (s SensitiveString) LogValue() slog.Value {
	return slog.StringValue("[REDACTED]")
}

// MaskedJSON 用于结构体，按需暴露字段
type MaskedJSON struct {
	Data any
}

func (m MaskedJSON) LogValue() slog.Value {
	// 可在此实现更复杂的脱敏逻辑
	return slog.AnyValue(m.Data)
}
```

## 3. 在 RAGAgent 中初始化与使用

```go
package main

import (
	"context"
	"os"

	"ragagent/pkg/logger"
)

func main() {
	// 配置：JSON 格式、同时写入文件与标准输出、采样 10%
	log := logger.New(
		logger.WithLevel(logger.LevelInfo),
		logger.WithFormat(logger.FormatJSON),
		logger.WithMultiWriter(
			slog.NewJSONHandler(os.Stdout, nil),  // 控制台
			slog.NewJSONHandler(mustOpenFile("/var/log/ragagent.log"), nil), // 文件
		),
		logger.WithSampling(0.1),
		logger.WithContextExtractor(otelContextExtractor),
		logger.WithReplaceAttr(func(groups []string, a slog.Attr) slog.Attr {
			// 将所有键名统一为 snake_case（示例）
			return slog.Attr{Key: toSnake(a.Key), Value: a.Value}
		}),
	)

	// 使用
	ctx := context.Background()
	log.InfoContext(ctx, "RAGAgent starting", "version", "1.0.0")

	err := someOperation()
	if err != nil {
		log.ErrorContext(ctx, err, "operation failed", "component", "retriever")
	}
}
```

## 4. 最佳实践映射

| 最佳实践                     | 在本封装中的体现                                                                 |
|------------------------------|----------------------------------------------------------------------------------|
| 标准化日志接口               | 通过 `LogValuer` 实现敏感字段过滤，`SensitiveString` 等类型确保一致性            |
| 错误日志添加堆栈             | `stacktraceMiddleware` 自动为 `Error` 级别的记录附加完整堆栈                     |
| Lint 规则（sloglint）        | 配置通过外部 linter 强制键名风格和参数一致性，包装器本身不干涉                    |
| 集中管理但先持久化到本地文件 | `WithFileOutput` / `WithMultiWriter` 将日志同时写入本地文件，与 stdout 解耦      |
| 日志采样                     | `WithSampling(rate)` 集成 `slog-sampling`，降低高吞吐下的成本                    |
| 使用日志管理服务             | 封装不绑定特定服务，通过标准输出或文件可轻松对接 Vector/Fluentd/Better Stack 等 |

## 5. 附加建议

- **生产环境**：务必配置日志轮转（如 `lumberjack`），避免磁盘写满。
- **上下文抽取**：结合 RAGAgent 的请求链，可注入 `request_id`、`user_id` 等业务字段。
- **性能**：采样、异步写入（可使用 `slogmulti.Pipe` 结合缓冲 channel）在高并发下尤其重要。

这套封装在保证性能的同时，将日志提升到“可观测性一等公民”的地位，是 RAGAgent 全链路追踪与快速排障的坚实基础。
